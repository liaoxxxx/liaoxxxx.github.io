关于两种连接mysql的方式,面向对象/面向过程;

茴香豆的茴字的N种写法.

  • 面向对象的方式连接mysql数据库. 

    
                    $conn=new mysqli("localhost","root","root");    //建立连接;

                    $conn->select_db("my_db");                        //选择database;

                    $result = $conn->query('SELECT * FROM user');   //查询数据,返回结果;

                    $result_arr=$result->fetch_array();             //获得返回结果的记录;

                    print_r($result_arr);                             //输出 $result_arr;

                    $result->free_result();                           //释放 查询结果 所占用的运行内存;

                    $conn->close();                                    //关闭连接进程;

  • 面向过程的方式连接mysql数据库. 

    
                     //建立连接;
                    $conn=mysqli_connect("localhost","root","root");

                    //选择database;
                    mysqli_select_db($conn,"my_db");

                    //查询数据,返回结果;
                    mysqli_query($conn,'select * from user');

                    //获得返回结果的记录;
                    $fetch_arr= mysqli_fetch_array($result);

                    //输出 $result_arr;
                    print_r($fetch_arr);

                    //释放 查询结果 所占用的运行内存;
                    mysqli_free_result($result);

                    //关闭连接进程;
                    mysqli_close($conn);

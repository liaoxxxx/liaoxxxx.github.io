	{
		if (!$_FILES[$uploadTagName]['error'])                                                   //判断是否上传成功；
		{

			$type=$_FILES['files']['type'];
			$index=strripos($type,'/');

			$type=substr($type,$index+1);                                                        //获得文件类型;

			if(0)                                                                               // $type

			}

			if (!file_exists($pathName))                                                        //判断路径是否不存在;
			{
				mkdir($pathName);                                                               //不存在该路径则创建该路径;
			}

			$fileName=md5(date("Y-m-d-H:i:s").mt_rand(10000,99999));                            //mt_rand()为避免文件重名

			copy($_FILES[$uploadTagName]['tmp_name'],$pathName.'/'.$fileName.'.'.$type);        //移动该文件;

			echo "上传成功";
			return  $pathName.'/'.$fileName.'.'.$type;
		}
		else
		{

			echo "'上传失败'";
                return false;
              }

        function download($files)
    {
	    $files="upload/image/2018-06-27/00.jpg";                                         //路径；
	    $index=strripos($files,"/");
	    $fileName=substr($files,$index+1);                                              //获得文件名；
	    $size=filesize($files);                                                         //获取文件大小；
	    header("Content-type:application/octet-stream");
	    header("Content-Disposition:attachment;filename='$fileName' ");                  //给下载的文件命名；
	    header("Content-Length:{$size}");                                               //注明文件尺寸；
	    header("Content-Transfer-Encoding:binary");
	    readfile($files);
    }

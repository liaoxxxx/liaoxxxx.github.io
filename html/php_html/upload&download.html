                        function uploadFileLot($tagName,$regex,$path)
{
    //new array(); 用于返回图片路径
    $pathArray=array();
    $time=time();
    $path.=date("Y-m-d",$time);
    //判断路径是否不存在;
    if (!file_exists($path))
    {
        //不存在该路径则创建该路径;
        mkdir($path);
    }
    //获得文件数量
    $filesCount=count($_FILES[$tagName]['name']);
    for ($i=0;$i<$filesCount;$i++)
    {
        //返回的 例如 image/jpeg,即jpeg文件;
        $type= $_FILES[$tagName]['type'][$i];
        $result= preg_match($regex,$type);
        //验证文件格式
        if (!$result)
        {
            echo'<script>alert( "文件格式不支持")</script>';
            die;
        }
        //把文件类型修剪为 例如 image/jpeg 获得jpeg;
        $type=substr($type,strripos($type,'/')+1);
        //mt_rand()为避免文件重名
        $fileName=md5(mt_rand(1000,9999).mt_rand(10000,99999));
        //移动该文件;
        $copyRe= copy($_FILES[$tagName]['tmp_name'][$i],$path.'/'.$fileName.'.'.$type);
        if ($copyRe)
        {
            $pathArray[$i]= $path.'/'.$fileName.'.'.$type;
        }
        else
        {
            echo'<script>alert("copy2pathFailed")</script>';
            die();
        }
    }
    echo'<script>alert("文件上传成功")</script>';
    return $pathArray;
}

    //路径； $files="upload/image/2018-06-27/00.jpg";
    function download($files)
    {
	    $index=strripos($files,"/");
	    $fileName=substr($files,$index+1);                                              //获得文件名；
	    $size=filesize($files);                                                         //获取文件大小；
	    header("Content-type:application/octet-stream");
	    header("Content-Disposition:attachment;filename='$fileName' ");                  //给下载的文件命名；
	    header("Content-Length:{$size}");                                               //注明文件尺寸；
	    header("Content-Transfer-Encoding:binary");
	    readfile($files);
    }
